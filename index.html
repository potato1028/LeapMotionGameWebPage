class RulesEngine:
    # ... (초기화 생략) ...

    # [핵심] 주사위 판정 로직
    # Advantage 적용 (주사위 2개 중 높은 값) + 히든 보너스
    def roll_check(self, stat_name: str, difficulty_check: int) -> dict:
        modifier = self._get_modifier(stat_name)
        
        # Advantage System
        roll1 = random.randint(1, 20)
        roll2 = random.randint(1, 20)
        roll = max(roll1, roll2)
        
        # Hidden Gameplay Bonus (+5)
        hidden_bonus = 5
        total = roll + modifier + hidden_bonus
        
        is_success = total >= difficulty_check
        outcome = "success" if is_success else "failure"
        
        return {
            "check_type": stat_name,
            "roll": roll,
            "total": total,
            "outcome": outcome
        }

    # AI Tool Definition (Schema)
    # AI가 이 정보를 보고 스스로 함수를 호출함
    def get_tools_schema():
        return [
            { 
                "type": "function", 
                "function": { 
                    "name": "roll_check", 
                    "description": "플레이어 행동 판정 수행", 
                    "parameters": { ... } 
                } 
            }
        ]

// [핵심] 서버 응답 처리 및 UI 업데이트
IEnumerator SendActionRequest(string action)
{
    // ... (HTTP 요청 생성) ...

    using (UnityWebRequest request = new UnityWebRequest(serverUrl, "POST"))
    {
        yield return request.SendWebRequest();

        if (request.result == UnityWebRequest.Result.Success)
        {
            ActionResponse response = 
                JsonUtility.FromJson(request.downloadHandler.text);
            
            // 1. AI GM의 서술(Narration) 출력
            AddChatMessage("GM (AI)", response.narration);

            // 2. 툴 호출 결과(주사위) 시각화
            if (response.last_tool_response != null)
            {
                UpdateDiceResultUI(response.last_tool_response);
            }
        }
    }
}

// 주사위 패널 UI 업데이트
void UpdateDiceResultUI(ToolResponse tool)
{
    diceResultPanel.SetActive(true);
    checkNameText.text = tool.check_type + " 판정";
    diceValueText.text = tool.total.ToString();
    
    if (tool.outcome == "success") {
        resultText.text = "SUCCESS";
        resultText.color = Color.green;
    } else {
        resultText.text = "FAILURE";
        resultText.color = Color.red;
    }
}
